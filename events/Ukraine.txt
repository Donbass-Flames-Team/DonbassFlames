#dpr uprising
country_event = {
	id = ukraine.2
	title = ukraine.2.t
	desc = ukraine.2.d
	picture = [GetHitlerHandshakeEventPicture]
	
	is_triggered_only = yes

	immediate = { 
		hidden_effect = {
			DPR = {
				declare_war_on = {
                    target = UKR
                    type = annex_everything
                }
			}
			if = {
				limit = {
					check_variable = {
                        var = DonetskSupplyValue
                        value = 1
                        compare = equals
                    }
				}
				DPR = {
		            add_equipment_to_stockpile = { 
				        type = infantry_equipment_1 
				        amount = 1500 
			        }
				}
			}
			if = {
				limit = {
					check_variable = {
                        var = DonetskSupplyValue
                        value = 2
                        compare = equals
                    }
				}
				DPR = {
		            add_equipment_to_stockpile = { 
				        type = infantry_equipment_1 
				        amount = 2500 
			        }					
				}
			}
			if = {
				limit = {
					check_variable = {
                        var = DonetskSupplyValue
                        value = 3
                        compare = greater_than_or_equals
                    }
				}
				DPR = {
		            add_equipment_to_stockpile = { 
				        type = infantry_equipment_1 
				        amount = 4500 
			        }					
				}
			}
		}
	}
	
	option = {
		name = ukraine.2.a
	}

	option = {
		name = ukraine.2.b
		DPR = {
			change_tag_from = root
		}
	}
}

	desc = ukraine.19.d
	#picture = GFX_report_event_royal_parade
	
	is_triggered_only = no
	fire_only_once = yes

        tag = UKR

            value = 50
            compare = greater_than_or_equals
        }
    }

	immediate = {
        hidden_effect = {
		    add_to_variable = {
                var = UKR_radicalisation_of_proeuropeans_value

            }

                var = UKR_radicalisation_of_nationalists_value
                value = 2
            }

                var = UKR_radicalisation_of_prorussians_value

            }		
		}
	}

	option = {
		name = ukraine.19.a
		add_stability = -0.025
		custom_effect_tooltip = UKR_add_radicalisation_of_maidan  
	}
}
