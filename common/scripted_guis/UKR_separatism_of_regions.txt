       context_type = decision_category
	   window_name = "UKR_separatism"
 
        visible = {
            tag = UKR
        }
 
		effects = {
			UKR_separatism_button_click = {
				if = {
					limit = { has_variable = show_separatism_in_ukraine_map }

				}
				else = {
					set_variable = { show_separatism_in_ukraine_map = 1 }
				}
			}
		}

        triggers = {

        }
		
		properties = {
			
		}
    }

    UKR_separatism_map = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "UKR_separatism_map"
		dirty = show_separatism_in_ukraine_map

		visible = {
			check_variable = { show_separatism_in_ukraine_map > 0 }
		}
		
		effects = {
			close_map_button_click = {
				clear_variable = show_separatism_in_ukraine_map
		    }
			coat_donetsk_button_click  = {
				set_variable = { select_separatists_region = 1}
			}
			coat_luhansk_button_click = {
				set_variable = { select_separatists_region = 2}
			}
			coat_kharkiv_button_click = {
				set_variable = { select_separatists_region = 3}
			}
			coat_odesa_button_click = {
				set_variable = { select_separatists_region = 4}
			}
			coat_krym_button_click = {
				set_variable = { select_separatists_region = 5}
			}
			separatism_increase_button_click = {
				if = {
					limit = { check_variable = {select_separatists_region = 1} }
					
				}
				if = {
					limit = { check_variable = {select_separatists_region = 2} }
					
				}
				if = {
					limit = { check_variable = {select_separatists_region = 3} }
					
				}
				if = {
					limit = { check_variable = {select_separatists_region = 4} }
					
				}
				if = {
					limit = { check_variable = {select_separatists_region = 5} }
					
				}
			}
		}
		
		dynamic_lists = {

		}
		
	    properties = {
			separatism_gui_background = {
				image = "[separatism_gui_background_image]"
